Introduction au Langage C

Le C est un langage de programmation compilé, bas niveau, créé par Dennis Ritchie aux Laboratoires Bell en 1972. Il est à la base de nombreux systèmes d'exploitation (Linux, Unix, Windows) et langages modernes (C++, Java, Python).

Pourquoi apprendre le C ?

  • Contrôle total de la mémoire et des ressources
  • Extrêmement rapide et portable
  • Base indispensable pour comprendre d'autres langages
  • Utilisé en programmation système, embarqué, noyaux OS

Structure d'un Programme C

Tout programme C doit posséder une fonction main(). C'est le point d'entrée du programme.

// Inclusion d'une bibliothèque standard
#include <stdio.h>

/* Commentaire
   multi-ligne */
int main(void) {
    printf("Bonjour le monde!\n");
    return 0;  // 0 = succès
}

Chaînes de Caractères

En C, une chaîne est un tableau de char terminé par le caractère nul '\0'.

// Déclaration littérale (en lecture seule)
char *s1 = "Bonjour";        // ⚠ Non modifiable !

// Tableau — modifiable
char s2[20] = "Bonjour";      // réservé pour 20 chars
char s3[] = "Bonjour";        // taille = 8 (7 + '\0')

// Accès aux caractères
s2[0] = 'b';               // "bonjour"
printf("%c\n", s3[3]);     // 'j'

// Affichage
printf("%s\n", s2);          // %s pour une chaîne
puts(s2);                   // affiche + \n

// Longueur
int len = strlen(s2);       // 7 (sans '\0')
💡 sizeof(s3) retourne 8 (inclut '\0'), tandis que strlen(s3) retourne 7.

Introduction aux Pointeurs

Un pointeur est une variable qui stocke l'adresse mémoire d'une autre variable.

int x = 42;
int *p = &x;     // p pointe vers x
                  // &x = adresse de x

printf("%d\n", x);   // 42 — valeur de x
printf("%p\n", p);   // adresse ex: 0x7ffe3a4b2c10
printf("%d\n", *p);  // 42 — déréférencement

*p = 100;           // Modifie x via le pointeur
printf("%d\n", x);   // 100

Déclaration des pointeurs

int    *pi;    // Pointeur vers int
double *pd;    // Pointeur vers double
char   *pc;    // Pointeur vers char
void   *pv;    // Pointeur générique (pas de type)

// ⚠ Piège syntaxique
int *a, b;    // a est un pointeur, b est un int !
int *a, *b;   // a et b sont des pointeurs

// Pointeur NULL (ne pointe vers rien)
int *p = NULL;
if (p != NULL) { *p = 5; }  // Vérifier avant déréférencer
⚠️ Déréférencer un pointeur NULL ou non initialisé = comportement indéfini (crash). Toujours initialiser les pointeurs.

Arithmétique des Pointeurs

L'arithmétique des pointeurs permet de naviguer dans la mémoire en tenant compte de la taille du type pointé.

int tab[] = {10, 20, 30, 40, 50};
int *p = tab;   // pointe sur tab[0]

printf("%d\n", *p);      // 10
printf("%d\n", *(p+1));  // 20 (avance de sizeof(int) octets)
printf("%d\n", *(p+4));  // 50

p++;            // p pointe maintenant sur tab[1]
printf("%d\n", *p);      // 20

p += 2;         // p pointe sur tab[3]
printf("%d\n", *p);      // 40

// Différence entre pointeurs
int *debut = tab;
int *fin = tab + 4;
printf("%td\n", fin - debut);  // 4 (éléments entre les deux)

Équivalence tableau / pointeur

tab[i]  ≡  *(tab + i)   // identiques !
&tab[i] ≡  tab + i

Allocation Dynamique de Mémoire

L'allocation dynamique se fait sur le tas (heap), contrairement aux variables locales qui sont sur la pile (stack).

#include <stdlib.h>

// malloc — alloue n octets, non initialisé
int *p = (int *)malloc(5 * sizeof(int));
if (p == NULL) { /* Toujours vérifier ! */ }

// calloc — alloue et initialise à 0
int *q = (int *)calloc(5, sizeof(int));

// Utilisation
for (int i = 0; i < 5; i++) p[i] = i * 10;

// realloc — redimensionner un bloc existant
p = (int *)realloc(p, 10 * sizeof(int));
if (p == NULL) { /* l'ancien bloc est toujours valide */ }

// free — libérer la mémoire (obligatoire !)
free(p);
p = NULL;   // Bonne pratique : remettre à NULL après free

Exemple complet : tableau dynamique

#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    printf("Taille du tableau : ");
    scanf("%d", &n);

    int *tab = (int *)malloc(n * sizeof(int));
    if (!tab) { perror("malloc"); return 1; }

    for (int i = 0; i < n; i++) tab[i] = i * i;
    for (int i = 0; i < n; i++)
        printf("%d\n", tab[i]);

    free(tab);
    return 0;
}

Pointeurs de Fonctions

// Syntaxe : type_retour (*nom)(paramètres)
int additionner(int a, int b) { return a + b; }
int multiplier(int a, int b) { return a * b; }

int main() {
    // Déclarer et assigner
    int (*operation)(int, int);
    operation = additionner;
    printf("%d\n", operation(3, 4));   // 7

    operation = multiplier;
    printf("%d\n", operation(3, 4));   // 12
}

typedef pour simplifier

typedef int (*OperFunc)(int, int);

void appliquer(OperFunc f, int x, int y) {
    printf("Résultat : %d\n", f(x, y));
}

appliquer(additionner, 10, 5);  // 15
appliquer(multiplier, 10, 5);  // 50

struct — Structures

Une struct regroupe plusieurs variables de types différents sous un même nom.

// Déclaration d'une structure
struct Etudiant {
    char  nom[50];
    int   age;
    float moyenne;
};

// Création d'une variable
struct Etudiant e1;

// Initialisation
struct Etudiant e2 = {"Alice", 20, 17.5f};

// Accès aux membres avec .
e1.age = 22;
strcpy(e1.nom, "Bob");
e1.moyenne = 14.0f;

printf("%s a %d ans (moyenne: %.1f)\n",
       e2.nom, e2.age, e2.moyenne);

Pointeur vers une structure (opérateur →)

struct Etudiant *pe = &e2;

// Ces deux formes sont équivalentes :
(*pe).age = 21;
pe->age   = 21;   // Syntaxe avec flèche (préférée)

// Allocation dynamique d'une structure
struct Etudiant *pe2 = malloc(sizeof(struct Etudiant));
pe2->age = 19;
free(pe2);

Passer une structure à une fonction

// Par valeur — copie complète (coûteux pour grandes structs)
void afficherEtudiant(struct Etudiant e) { /* ... */ }

// Par pointeur — recommandé (pas de copie)
void modifierAge(struct Etudiant *e, int nouvelAge) {
    e->age = nouvelAge;
}

// Par pointeur const — lecture seule, pas de copie
void afficher(const struct Etudiant *e) {
    printf("%s\n", e->nom);
}

union — Mémoire Partagée

Une union permet à plusieurs membres de partager le même espace mémoire. La taille = taille du plus grand membre.

union Donnee {
    int    entier;
    float  reel;
    char   caractere;
};

union Donnee d;
d.entier = 42;
printf("%d\n", d.entier);       // 42

d.reel = 3.14f;
printf("%.2f\n", d.reel);      // 3.14
printf("%d\n", d.entier);       // ⚠ Valeur indéfinie !

printf("%zu\n", sizeof(d));    // 4 (taille du plus grand membre)

Union avec tag (pattern sûr)

typedef enum { TYPE_INT, TYPE_FLOAT, TYPE_STR } TypeTag;

typedef struct {
    TypeTag tag;
    union {
        int    i;
        float  f;
        char  *s;
    } valeur;
} Variant;

Variant v = {TYPE_FLOAT, {.f = 3.14f}};
if (v.tag == TYPE_FLOAT)
    printf("%.2f\n", v.valeur.f);
